İletişim özelliğini kullanabilmek için:

Ana dizine .env dosyası oluştur içeriğini alttaki gibi yap

# Django
SECRET_KEY=SECRET_KEY
DEBUG=True
ALLOWED_HOSTS=127.0.0.1,localhost

# E-posta
EMAIL_HOST=smtp.gmail.com
EMAIL_PORT=587
EMAIL_USE_TLS=True
EMAIL_HOST_USER=asd@gmail.com
EMAIL_HOST_PASSWORD=uygulama_sifresi
DEFAULT_FROM_EMAIL=asd@gmail.com


# Telegram Bot
TELEGRAM_BOT_TOKEN=TelegramToken
